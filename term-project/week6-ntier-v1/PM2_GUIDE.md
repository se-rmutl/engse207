### 5.2 Start Options

```bash
# Start with specific options
pm2 start server.js \
    --name "taskboard-api" \        # аёҠаё·а№Ҳаёӯ process
    --watch \                        # Auto-restart а№ҖаёЎаё·а№Ҳаёӯа№„аёҹаёҘа№Ңа№ҖаёӣаёҘаёөа№Ҳаёўаёҷ
    --ignore-watch="node_modules" \  # а№„аёЎа№Ҳ watch folders аёҷаёөа№ү
    --max-memory-restart 200M \      # Restart аё–а№үаёІ memory > 200MB
    --env production \               # Set NODE_ENV
    --log /var/log/app.log \         # Log file path
    --time \                         # Add timestamp to logs
    --cron-restart="0 0 * * *" \     # Restart аё—аёёаёҒа№Җаё—аёөа№ҲаёўаёҮаё„аё·аёҷ
    --no-autorestart                 # аёӣаёҙаё” auto-restart
```

### 11.4 Graceful Shutdown Best Practices

```javascript
// аё•аёұаё§аёӯаёўа№ҲаёІаёҮаё—аёөа№Ҳаё–аё№аёҒаё•а№үаёӯаёҮ
const server = app.listen(PORT, () => {
    console.log(`Server running on port ${PORT}`);
    
    // аёҡаёӯаёҒ PM2 аё§а№ҲаёІ ready
    if (process.send) {
        process.send('ready');
    }
});

// Handle graceful shutdown
const shutdown = async (signal) => {
    console.log(`${signal} received. Graceful shutdown...`);
    
    // 1. аё«аёўаёёаё”аёЈаёұаёҡ connection а№ғаё«аёЎа№Ҳ
    server.close(async () => {
        console.log('HTTP server closed');
        
        // 2. аёӣаёҙаё” database connections
        await db.close();
        console.log('Database connections closed');
        
        // 3. Exit
        process.exit(0);
    });
    
    // Force exit аё«аёҘаёұаёҮ 10 аё§аёҙаёҷаёІаё—аёө
    setTimeout(() => {
        console.error('Force shutdown');
        process.exit(1);
    }, 10000);
};

process.on('SIGTERM', () => shutdown('SIGTERM'));
process.on('SIGINT', () => shutdown('SIGINT'));
```
